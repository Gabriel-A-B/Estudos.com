--------------------------------------------------------------

        Idade

            p
                label for="iano" Nacimento /label
                input type="number" name="ano" id="iano" 
                requirer oninput="calcIdade()"
            /p

            p
                label for="iIdade" Sua Idade /label
                output id="iIdade" 0 /output
            /p

    script
        function calcIdade() {
            let atual = new Date().getFullYear()
            iIdade.innerHTML = Number(atual) - 
            Number(iano.value)
        }
    /script
